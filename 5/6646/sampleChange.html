def to_dict(config):
    if type(config) is list:
        return [to_dict(c) for c in config]
    elif type(config) is EasyDict:
        return dict([(k, to_dict(v)) for k, v in config.items()])
    else:
        return config


def dump_config(config):
    config = to_dict(config)
